; FriendlyName; SourceUrl; App1, App2; ParameterForPath; IsFullPath; DefaultPath; DefaultFileName
;

; ParameterForPath = the parameter that needs to be set within redink.ini
; IsFullPath = True if the parameter holds the full path; False if only the directory
; DefaultPath = the default path to be used if none is use (will be expanded)
; DefaultFileName = the default filename of the file on the target
